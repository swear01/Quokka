% Noto Sans CJK TC: ~/.local/share/fonts/noto-sans-cjk-tc/*.otf (build.sh installs/extracts)
\newcommand{\CJKFontDir}{/home/swear01/.local/share/fonts/noto-sans-cjk-tc/}
\newcommand{\CJKFontRegularOTF}{\CJKFontDir NotoSansCJKtc-Regular.otf}
\IfFileExists{\CJKFontRegularOTF}{%
  \setCJKmainfont{NotoSansCJKtc-Regular}[
    Path=\CJKFontDir,
    Extension=.otf,
    BoldFont=NotoSansCJKtc-Bold,
  ]
  \setCJKsansfont{NotoSansCJKtc-Regular}[
    Path=\CJKFontDir,
    Extension=.otf,
    BoldFont=NotoSansCJKtc-Bold,
  ]
  \setsansfont{NotoSansCJKtc-Regular}[
    Path=\CJKFontDir,
    Extension=.otf,
    BoldFont=NotoSansCJKtc-Bold,
  ]
}{%
  \setCJKmainfont{Noto Sans CJK TC}[
    BoldFont={Noto Sans CJK TC Bold},
  ]
  \setCJKsansfont{Noto Sans CJK TC}[
    BoldFont={Noto Sans CJK TC Bold},
  ]
  \setsansfont{Noto Sans CJK TC}[
    BoldFont={Noto Sans CJK TC Bold},
  ]
}
